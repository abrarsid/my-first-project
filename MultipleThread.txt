package com.company;

class SycroDeno
{
    void printTable(int n)
    {
        for (int i=1;i<=10;i++)
        {
            System.out.println(n*i);
            try {
                Thread.sleep(1000);
            }catch(Exception e)
            {
                e.printStackTrace();
            }
        }
    }

}
class Table extends Thread
{ SycroDeno sd;
    public Table(SycroDeno sd)
    {
       this.sd=sd;
    }

    @Override
    public void run()
    {
       sd.printTable(2);
    }
}
